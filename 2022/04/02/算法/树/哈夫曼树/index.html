主要是把字符串读到vector中，然后记录每个字符出现的次数，然后构建哈夫曼树，然后再生成哈夫曼编码，再写入。

#include <iostream>
#include <Windows.h>
#include <fstream>
#include <vector>
#include<stdio.h>
#include <algorithm>
#include <cstring>
#include <map> 
#include <bitset>
#include<unordered_map>
using namespace std;
long byteNum = 0;
typedef struct
{
    int weight;
    int parent, lchild, rchild;
}HafuNode, * HufumanTree;
typedef struct
{
    char* data;
    int* num;
    int length;
}TNode;
typedef struct
{
    char* data;
    char** HM;
}Code;
typedef char** HuffmanCode;




class Hafuman
{
private:
    HuffmanCode code;
    TNode tnod;
    HufumanTree hafutree;
    map<string, char> hafumanHash;
    //存文件内容
    vector<char> str;
public:
    void compression(string decomFile,string comFile)
    {
        Read(comFile);
        TNodeCount();
        CreateHuffmanTree();
        CreatHuffmanCode();
        int i, j, k;
        unsigned int tmp = 0;
        int bit = 0;
        ofstream outfile("compression.txt", ios::out);
        ofstream outComFile(decomFile, ios::out);
        if (!outfile)
        {
            cerr << "open error" << endl;
            exit(1);
        }
        //写入编码
        outComFile << tnod.length << ' ';
        for (i = 0; i < tnod.length; i++)
        {
            outComFile << tnod.data[i];
            outComFile << code[i + 1];
            outComFile << ' ';
        }

        for (i = 0; i < str.size(); i++)
        {
            for (j = 0; j < tnod.length; j++)
            {
                if (tnod.data[j] == str[i])
                {
                    break;
                }
            }
            for (k = 0; code[j + 1][k] != '\0'; k++)
            {
                outfile << code[j + 1][k];
                if (code[j + 1][k] == '0')
                {
                    tmp = tmp | 0;
                }
                else
                {
                    tmp = (tmp | 1);
                }
                bit = (bit + 1) % 32;
                if (!bit)
                {
                    outComFile << tmp << ' ';
                    tmp = 0;
                }
                else
                    tmp = tmp << 1;
                byteNum++;
            }
        }
        tmp = tmp << (32 - byteNum % 32 - 1);
        outComFile << tmp << ' ';
        outComFile << byteNum;
        cerr << "文件的总的字符数为 " << byteNum << endl;
        outfile.close();
        //写总共的位数


        cerr << "压缩成功!,可以到compression.txt中查看具体二进制码，压缩文件为 "<<decomFile << endl;


        //outComFile.seekg(0, ios::beg);

        //streampos size = outComFile.tellg();
        cout << "压缩文件大小为：" << byteNum/8 << " 字节" << endl;
        outComFile.close();

    }
    void decompression(string decomFilename, string comfilename)
    {
        char a[30];
        ofstream outfile(comfilename, ios::out);
        ifstream inComfile(decomFilename, ios::in);
        if (!outfile)
        {
            cerr << "open error" << endl;
            exit(1);
        }
        if (!inComfile)
        {
            cerr << "open error" << endl;
            exit(1);
        }
        int bytenum;
        inComfile.seekg(-2L, ios::end);
        inComfile >> bytenum;

        inComfile.seekg(0, ios::beg);
        int codeNum;
        inComfile >> codeNum;

        char* code = new char[100];
        char ch;
        unordered_map<string, char> hash;
        for (int i = 0; i < codeNum; i++)
        {
            inComfile >> ch;
            inComfile >> code;
            string tmp = code;
            hash[tmp] = ch;
        }
        unsigned int m;
        string mystr = "";
        for (; bytenum >= 32; bytenum -= 32)
        {
            inComfile >> m;
            unsigned int mask = 0x80000000;
            int countForByte = 32;

            while (1)
            {
                if (!(mask & m))
                    mystr.append("0");
                else
                    mystr.append("1");
                m = m << 1;
                countForByte--;
                if (hash.count(mystr))
                {
                    outfile << hash[mystr];
                    mystr = "";
                }
                if (countForByte <= 0)
                    break;
            }
        }
        inComfile >> m;
        unsigned int mask = 0x80000000;
        int countForByte = bytenum;

        while (1)
        {
            if (!(mask & m))
                mystr.append("0");
            else
                mystr.append("1");
            m = m << 1;
            countForByte--;
            if (hash.count(mystr))
            {
                outfile << hash[mystr];
                mystr = "";
            }
            if (countForByte <= 0)
                break;
        }
        cout << "解压成功！解压文件为" << comfilename<<endl;
        delete code;
        code = nullptr;

    }
    void CreatHuffmanCode()
    {
        int n = tnod.length;
        int pare, child, start;
        code = new char* [n + 1];
        char* cd = new char[n];
        cd[n - 1] = '\0';
        for (int i = 1; i <= n; i++)
        {
            start = n - 1;
            child = i;
            pare = hafutree[i].parent;
            while (pare != 0)
            {
                start--;
                if (child == hafutree[pare].lchild)
                {
                    cd[start] = '0';
                }
                else
                {
                    cd[start] = '1';
                }
                child = pare;
                pare = hafutree[child].parent;
            }
            code[i] = new char[n - start];
            strcpy(code[i], &cd[start]);
        }
        for (int i = 1; i <= n; i++)
        {
            cerr << tnod.data[i-1] << "权重 : " << tnod.num[i-1];
            cerr <<"二进制编码为"<< code[i] << endl;
        }
        delete cd;
    }
    void initTnode()
    {
        tnod.data = new char[256];
        tnod.num = new int[256];
        if (tnod.data == NULL || tnod.num == NULL)
        {
            cout << "发生错误" << endl;
            exit(1);
        }
        tnod.length = 0;
    }
    void Read( string filename)
    {
        char ch;
        ifstream infile(filename, ios::in);
        if (!infile)
        {
            cout << "open error" << endl;
            exit(1);
        }
        while (infile.peek() != EOF)
        {
            infile.get(ch);
            str.push_back(ch);
        }
        infile.seekg(0, ios::end);
        streampos size = infile.tellg();
        cout << "源文件大小为：" << size << " 字节" << endl;
        infile.close();
    }
    bool find(const char ch, TNode t)
    {
        for (int i = 0; i < t.length; i++)
        {
            if (t.data[i] == ch)
            {
                return true;
            }
        }
        return false;
    }
    void TNodeCount()
    {
        int m = str.size(), j = 0;
        char ch;
        for (int i = 0; i < m; i++)
        {
            ch = str[i];
            if (!find(ch, tnod))
            {
                tnod.data[j] = ch;
                tnod.num[j] = count(str.begin(), str.end(), ch);
                tnod.length++;
                j++;
            }
        }
    }
    void Select(HufumanTree& tree, int a, int& b, int& c)
    {
        int min1, min2, minweight = 10000;
        for (int i = 1; i <= a; i++)
        {
            if (tree[i].parent == 0)
            {
                if (tree[i].weight < minweight)
                {
                    minweight = tree[i].weight;
                    min1 = i;
                }
            }
        }
        tree[min1].parent = 1;
        minweight = 10000;
        for (int i = 1; i <= a; i++)
        {
            if (tree[i].parent == 0)
            {
                if (tree[i].weight < minweight)
                {
                    minweight = tree[i].weight;
                    min2 = i;
                }
            }
        }
        tree[min2].parent = 1;
        b = min1;
        c = min2;
    }
    void CreateHuffmanTree()
    {
        int n = tnod.length;
        if (n <= 1)
        {
            return;
        }
        int m = 2 * n - 1;
        hafutree = new HafuNode[m + 1];
        for (int i = 1; i <= m; i++)//为0表示没有左右节点，父节点
        {
            hafutree[i].lchild = 0;
            hafutree[i].parent = 0;
            hafutree[i].rchild = 0;
        }
        for (int i = 1; i <= n; i++)
        {
            hafutree[i].weight = tnod.num[i - 1];
        }
        int s1, s2;
        for (int i = n + 1; i <= m; i++)
        {
            Select(hafutree, i - 1, s1, s2);
            hafutree[s1].parent = i;
            hafutree[s2].parent = i;
            hafutree[i].lchild = s1;
            hafutree[i].rchild = s2;
            hafutree[i].weight = hafutree[s1].weight + hafutree[s2].weight;
        }
    }
};

股票买卖问题1

当前的最大收益只依赖于之前的最小买入价格。只需要记住到目前为止的最小值，然后可以得到每个位置的收益。O(n)

class Solution {
public:
    int maxProfit(vector<int>& prices) 
    {
        int max=0;
        int len=prices.size();
        if(len==0)
            return 0;
        int min=prices[0];
        for(int i=0;i<len;i++)
        {
            if(prices[i]>min)
            {
                int t=prices[i]-min;
                if(max<t)
                    max=t;
            }
            else
            {
                min=prices[i];
            }
        }
        return max;
    }
};

股票买卖问题2

这个题目朴素的想法就是找到每个峰谷和峰顶，然后求差异获得利润。
但是由于可以进行多次交易，那么只要明天比今天价格高就有得赚，就可以进行交易。不需要去找波峰波谷，因为day2-day1+day3-day2 == day3-day1

class Solution {
public:
    int maxProfit(vector<int>& prices) 
    {
        int res=0;
        int len=prices.size();
        for(int i=1;i<len;i++)
        {
            if(prices[i]>prices[i-1])
                res+=prices[i]-prices[i-1];
        }
        return res;
    }
};

股票买卖问题3

int maxProfit(vector<int>& prices) 
{
    int len=prices.size();
    if(len==0)
        return 0;
    vector<int> res1(3,0);
    vector<int> res2(3,0);
    for(int i=0;i<3;i++)
        res2[i]=-prices[0];
    for(int i=1;i<len;i++)
    {
        res2[0]=max(res2[0],-prices[i]);
        for(int j=1;j<=2;j++)
        {
            res1[j]=max(res1[j],res2[j-1]+prices[i]);
            res2[j]=max(res1[j]-prices[i],res2[j]);
        }
    }
    return res1[2];
}

股票买卖问题4

买卖k次

当前处于第几天；
已经交易的次数；
手头是否持有股票；
即根据手头是否持有股票，我们定义两个二维数组来定义状态：

dp0[i][j]: 第i天结束，已有j次买卖，手头没有股票时的最大利润
dp1[i][j]: 第i天结束，已有j次买卖，手头有股票时的最大利润
因此，dp0[0][j]对于所有j都要初始化为0，而dp1[0][j]对于所有j都要初始化为-prices[i]。如果我们将dp0所有值都求出来了，那么很明显dp0[n-1][k]就是在最后一天结束时已进行k次交易且手头无股票时的最大收益，也即返回结果。 先看初始状态:

当i==0 && j>=0: dp0[0][j] = 0, dp1[0][j] = -prices[0];
当i>0 && j==0: dp0[i][0] = 0, dp1[i][0] = max(dp1[i-1][0], -prices[i]);
再来考虑状态转移方程，当i>0且j>0时有

dp0[i][j] = max(dp0[i-1][j], dp1[i-1][j-1] + prices[i]) # 保持 or 卖出
dp1[i][j] = max(dp1[i-1][j], dp0[i-1][j] - prices[i]) # 保持 or 买入
有了状态定义及转移方程，剩下就好办了。

int maxProfit(vector<int>& prices) 
{
    int k=2;//可以改变k值
    int len=prices.size();
    if(len==0)
        return 0;
    vector<int> a(k+1,0);
    vector<vector<int>> res1(len,a);//没有股票在手中
    vector<vector<int>> res2(len,a);//有股票在手中
    int min=prices[0];
    for(int i=0;i<k;i++)
        res2[0][i]=-prices[0];
    for(int i=1;i<len;i++)
    {
        res2[i][0]=max(res2[i-1][0],-prices[i]);
    }
    for(int i=1;i<len;i++)
    {
        for(int j=1;j<=k;j++)
        {
            res1[i][j]=max(res1[i-1][j],res2[i-1][j-1]+prices[i]);
            res2[i][j]=max(res1[i-1][j]-prices[i],res2[i-1][j]);
        }
    }
    return res1[len-1][k];
}

但是这个效率不高，可以知道，当k>2/n的时候。就是买卖无数次了。
并且可以进行空间优化。

int maxProfit(int k, vector<int>& prices) 
{
    int len=prices.size();
    if(len==0||k==0)
        return 0;
    if(k>len/2)
    {
        int result=0;
        for(int i=1;i<len;i++)
        {
            if(prices[i]>prices[i-1])
                result+=prices[i]-prices[i-1];
        }
        return result;
    }
    vector<int> res1(k+1,0);
    vector<int> res2(k+1,0);
    int min=prices[0];
    for(int i=0;i<k;i++)
        res2[i]=-prices[0];
    for(int i=1;i<len;i++)
    {
        res2[0]=max(res2[0],-prices[i]);
        for(int j=1;j<=k;j++)
        {
            res1[j]=max(res1[j],res2[j-1]+prices[i]);
            res2[j]=max(res1[j]-prices[i],res2[j]);
        }
    }
    return res1[k];
}

股票买卖问题5

这里需要注意的就是当不持股，不在冷冻期的可能就是，上一个是冷冻期，然后这一天啥也没干，还有可能就是直接就是上一天。

int maxProfit(vector<int>& prices) 
{
    int len=prices.size();
    if(len==0)
        return 0;
    vector<int> res1(len,0);//不持股，不在冷冻期
    vector<int> res2(len,0);//持股，不在冷冻期
    res2[0]=-prices[0];
    vector<int> res3(len,0);//不持股，在冷冻期
    for(int i=1;i<len;i++)
    {
        res1[i]=max(res1[i-1],res3[i-1]);
        res2[i]=max(res1[i-1]-prices[i],res2[i-1]);
        res3[i]=res2[i-1]+prices[i];
    }
    return max(res3[len-1],res1[len-1]);
}
优化空间
    int maxProfit(vector<int>& prices) 
{
    int len=prices.size();
    if(len==0)
        return 0;
    int res1=0;//不持股，不在冷冻期
    int res2=-prices[0];//持股，不在冷冻期
    int res3=0;//不持股，在冷冻期
    for(int i=1;i<len;i++)
    {
        int a=res1;
        int b=res2;
        res1=max(a,res3);
        res2=max(a-prices[i],b);
        res3=b+prices[i];
    }
    return max(res3,res1);
}

股票买卖问题6

也是和之前的类似，选取两个状态，持有股票，不持有股票，在卖出时还需要减去手续费。

int maxProfit(vector<int>& prices, int fee) 
{
    int len=prices.size();
    int res1=0;
    int res2=-prices[0];
    int t;
    for(int i=0;i<len;i++)
    {
        t=res1;
        res1=max(res1,res2+prices[i]-fee);
        res2=max(res2,t-prices[i]);
    }
    return res1;
}

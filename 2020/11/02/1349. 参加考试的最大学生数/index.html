方法就是每次求前n排的人数和前n-1排的人数的关系，重点在于相邻两排斜着的人是否都有，一排是否有相邻的，是否坐在坏的座位上。
一一排除。遍历，所有的状态，一排最多2^8个状态。

class Solution {
public:
    int maxStudents(vector<vector<char>>& seats) 
    {
        int m=seats.size();
        int n=seats[0].size();
        vector<vector<int>> dp(m+1,vector<int>(1<<n)); 
        for(int i=1;i<=m;i++)
        {
            for(int j=0;j<(1<<n);j++)
            {
                bitset<8> bs(j);
                bool state=true;
                for(int k=0;k<n;k++)
                {
                    if((seats[i-1][k]=='#'&&bs[k])||(k<n-1&&bs[k]&&bs[k+1]))
                    {
                        state=false;
                        break;
                    }
                }
                if(state==false)
                {
                    dp[i][j]=-1;
                    continue;
                }
                for(int k=0;k<(1<<n);k++)
                {
                    if(dp[i-1][k]==-1)
                    {
                        continue;
                    }
                    bitset<8> lastbs(k);
                    bool state1=true;
                    for(int r=0;r<n;r++)
                    {
                        if(lastbs[r]&&(r>0&&bs[r-1]||r<n-1&&bs[r+1]))
                        {
                            state1=false;
                            break;
                        }
                    }
                    if(state1==true)
                    {
                        dp[i][j]=max(dp[i][j],dp[i-1][k]+int(bs.count()));
                    }
                }
            }
        }

        int result=0;
        for(int i=0;i<(1<<n);i++)
        {
            result=max(result,dp[m][i]);
        }
        return result;
    }
};
